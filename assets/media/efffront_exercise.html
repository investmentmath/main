# Cash rate
r=0.02

s1=0.08
sr1=0.05
s2=0.16
sr2=0.25
rho=-0.2
sigma=np.array([[s1**2,rho*s1*s2],[rho*s1*s2,s2**2]])
e=np.ones(2)

mu1=r+sr1*s1
mu2=r+sr2*s2
mu=np.array([mu1,mu2])

x=symbols('x')
expr=((A-2*B*x+C*x**2)/(A*C-B**2))**0.5

pl=plot_parametric(expr,x,(x,0.00,0.10),axis_center=(0.05,0.0),show=False)

pl.xlabel='Std. Dev.'
pl.ylabel='Exp. Ret.'
pl.title='Efficient Frontier (no risk-free asset)'
